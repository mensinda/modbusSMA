109 ErrorCode ModbusAPI::connect() {
110  auto logger = log::get();
111  if (mState != State::CONFIGURE) {
112  logger->error("ModbusAPI: can not connect() -- invalid object state '{}'", enum2Str::toStr(mState));
113  mState = State::ERROR;
114  return ErrorCode::INVALID_STATE;
115  }
116 
117  auto res = mConn->connect();
118  if (res != ErrorCode::OK) {
119  logger->error("ModbusAPI: unable to connect: '{}'", enum2Str::toStr(res));
120  mState = State::ERROR;
121  return res;
122  }
123 
124  logger->info("ModbusAPI: connected to {}", mConn->description());
125  mState = State::CONNECTED;
126  return ErrorCode::OK;
127 }

137 ErrorCode ModbusAPI::initialize() {
138  auto logger = log::get();
139  ErrorCode result;
140 
141  if (mState != State::CONNECTED) {
142  logger->error("ModbusAPI: can not initialize() -- invalid object state '{}'", enum2Str::toStr(mState));
143  mState = State::ERROR;
144  return ErrorCode::INVALID_STATE;
145  }
146 
147  // 1st: check database
148  if (!mDB->isConnected()) {
149  result = mDB->connect();
150  if (result != ErrorCode::OK) {
151  logger->error("ModbusAPI: DataBase initialization failed with {}", enum2Str::toStr(result));
152  mState = State::ERROR;
153  return result;
154  }
155  }
156 
157  mRegisters = make_shared<RegisterContainer>();
158  mRegisters->addRegisters(mDB->getRegisters("ALL"));
159  if (mRegisters->empty()) {
160  // This code should never be executed because of the DataBase validation
161  logger->error("ModbusAPI: No Registers in table 'ALL'");
162  mState = State::ERROR;
163  return ErrorCode::DATA_BASE_ERROR;
164  }
165 
166  // 2nd: set slave ID to 1
167  result = mConn->setSlaveID(1);
168 
169  if (result != ErrorCode::OK) {
170  mState = State::ERROR;
171  return result;
172  }
173 
174  // 3rd: request data
175  vector<uint16_t> rawData = mConn->readRegisters(42109, 4);
176 
177  logger->debug("Requesting basic inverter information:");
178 
179  if (rawData.size() != 4) {
180  logger->error("ModbusAPI: Failed to initialize -- requesting slave/unit ID failed");
181  mState = State::ERROR;
182  return ErrorCode::INITIALIZATION_FAILED;
183  }
184 
185  uint32_t serialNumber = (rawData[0] << 16) + rawData[1];
186  uint16_t susyID = rawData[2];
187  uint16_t unitID = rawData[3];
188 
189  logger->debug(" -- Physical serial number: {}", serialNumber);
190  logger->debug(" -- Physical SusyID: {}", susyID);
191  logger->debug(" -- Unit / Slave ID: {}", unitID);
192 
193  // 4th: set slave ID to unitID
194  result = mConn->setSlaveID(unitID);
195 
196  if (result != ErrorCode::OK) {
197  mState = State::ERROR;
198  return result;
199  }
200 
201  // 5th: determine the inverter type
202  rawData = mConn->readRegisters(30053, 2);
203  uint32_t inverterID = (rawData[0] << 16) + rawData[1];
204  logger->debug(" -- Inverter type ID: {}", inverterID);
205 
206  auto devices = mDB->getDeviceEnums();
207  bool found = false;
208  for (auto i : devices) {
209  if (i.id == inverterID) {
210  found = true;
211  mInverterType = i.name;
212  mInverterTypeID = inverterID;
213  mRegisters->addRegisters(mDB->getRegisters(i.table));
214  break;
215  }
216  }
217 
218  if (!found) {
219  logger->error("ModbusAPI: unknown inverter type {}", inverterID);
220  mState = State::ERROR;
221  return ErrorCode::INITIALIZATION_FAILED;
222  }
223 
224  logger->debug(" -- Inverter type: '{}'", mInverterType);
225  logger->debug(" -- Number of registers: {}", mRegisters->size());
226  logger->info("ModbusAPI: initialization for {} complete", mInverterType);
227  mState = State::INITIALIZED;
228  return ErrorCode::OK;
229 }

276 ErrorCode ModbusAPI::updateRegisters(vector<Register> _regList, size_t *_numUpdated) {
277  auto logger = log::get();
278  if (_numUpdated) { *_numUpdated = 0; }
279  if (mState != State::INITIALIZED) {
280  logger->error("ModbusAPI: updateRegisters() -- invalid object state '{}'", enum2Str::toStr(mState));
281  return ErrorCode::INVALID_STATE;
282  }
283  if (_regList.empty()) {
284  logger->warn("ModbusAPI: updateRegisters() -- empty register list ==> do nothing");
285  return ErrorCode::OK;
286  }
287 
288  sort(begin(_regList), end(_regList)); // Ensure that the list is sorted.
289 
290  // 1st: Create batches of registers.
291  std::vector<RegBatch> batches;
292  batches.reserve(_regList.size()); // Worst case: every register has its own batch
293  batches.push_back({0, {}, {}});
294 
295  for (Register i : _regList) {
296  RegBatch *curr = &batches.back();
297 
298  if (!curr->regs.empty()) {
299  if (((curr->size + i.size()) >= SMA_MODBUS_MAX_REGISTER_COUNT) || // Check if maximum request size is reached
300  ((curr->regs[0].reg + curr->size) != i.reg())) { // Check if continous
301  batches.push_back({0, {}, {}});
302  curr = &batches.back();
303  }
304  }
305 
306  curr->regs.push_back({i.reg(), curr->size, (uint16_t)i.size()});
307  curr->size += i.size();
308  }
309 
310  vector<uint16_t> singleRegData = {};
311  uint32_t counter = 1;
312  for (auto &i : batches) {
313  logger->debug("Fetching batch {} of {} -- Start: {}; Size: {}", counter++, batches.size(), i.regs[0].reg, i.size);
314  i.rawData = mConn->readRegisters(i.regs[0].reg, i.size);
315 
316  if (i.rawData.size() != i.size) {
317  logger->warn(
318  "ModbusAPI: updateRegisters() -- failed to fetch registers: Start = {}; Size = {}", i.regs[0].reg, i.size);
319  continue;
320  }
321 
322  for (auto j : i.regs) {
323  singleRegData.resize(j.size);
324  for (uint16_t k = 0; k < j.size; ++k) { singleRegData[k] = i.rawData[j.offset + k]; }
325  mRegisters->updateRegister(j.reg, singleRegData);
326  if (_numUpdated) { *_numUpdated += 1; }
327  }
328  }
329 
330  return ErrorCode::OK;
331 }
