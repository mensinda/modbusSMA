40 string trim(const string &str) {
41  size_t first = str.find_first_not_of(" \n\t");
42  if (string::npos == first) { return str; }
43  size_t last = str.find_last_not_of(" \n\t");
44  return str.substr(first, (last - first + 1));
45 }

58  vector<string> splitDesc = split(mDesc, '\n');
59  if (splitDesc.size() > 1) {
60  for (size_t i = 1; i < splitDesc.size(); ++i) {
61  vector<string> currSplit = split(splitDesc[i], '=');

64  string num = trim(currSplit[0]);
65  string name = currSplit[1];
66 
67  for (size_t j = 2; j < currSplit.size(); ++j) { name += "=" + currSplit[j]; }

70  uint32_t number;

72  number = (uint32_t)stoi(num);
73  } catch (...) { continue; }
74 
75  mEnumData[number] = name;
76  }

118 string Register::value() {
119  char *raw = reinterpret_cast<char *>(mData.data());
120  if (mType == DataType::STR32) {
121  if (mData.size() != 16) { return "Conversion ERROR: Invalid Data size!"; }
122  char data[33];
123  for (uint32_t i = 0; i < 32; i += 2) {
124  data[i + 0] = raw[i + 1];
125  data[i + 1] = raw[i + 0];
126  }
127  data[32] = '\0';
128  return data;
129  }
130 
131  if (mData == getNaN()) { return "NaN"; }
132 
133  uint32_t numDec = UINT32_MAX;
134 
135  switch (mFormat) {
136  case DataFormat::FW: {
137  if (mData.size() < 2) { return "Conversion ERROR: Invalid Data size!"; }
138  string suffix;
139  switch ((uint)raw[2]) {
140  case 0: suffix = "N"; break;
141  case 1: suffix = "E"; break;
142  case 2: suffix = "A"; break;
143  case 3: suffix = "B"; break;
144  case 4: suffix = "R"; break;
145  case 5: suffix = "S"; break;
146  default: suffix = to_string((uint)raw[3]); break;
147  }
148  return fmt::format("{}.{:0>2}.{:0>2}.{}", (uint)raw[1], (uint)raw[0], (uint)raw[3], suffix);
149  }
150 
151  case DataFormat::IP4:
152  case DataFormat::REV:
153  if (mData.size() < 2) { return "Conversion ERROR: Invalid Data size!"; }
154  return fmt::format("{}.{}.{}.{}", (uint)raw[0], (uint)raw[1], (uint)raw[2], (uint)raw[3]);
155 
156  case DataFormat::DT: {
157  time_t time = (time_t)valueUInt();
158  return asctime(localtime(&time));
159  }
160 
161  case DataFormat::ENUM: {
162  auto iter = mEnumData.find((uint32_t)valueUInt());
163  if (iter == end(mEnumData)) { return fmt::format("ENUM: {}", valueUInt()); }
164  return iter->second;
165  }
166 
167  case DataFormat::FIX0:
168  case DataFormat::Duration:
169  case DataFormat::HW:
170  case DataFormat::RAW:
171  case DataFormat::TM:
172  case DataFormat::FUNCTION_SEC: numDec = 0; [[fallthrough]];
173  case DataFormat::TEMP:
174  case DataFormat::FIX1: numDec = numDec == UINT32_MAX ? 1 : numDec; [[fallthrough]];
175  case DataFormat::FIX2: numDec = numDec == UINT32_MAX ? 2 : numDec; [[fallthrough]];
176  case DataFormat::FIX3: numDec = numDec == UINT32_MAX ? 3 : numDec; [[fallthrough]];
177  case DataFormat::FIX4:
178  numDec = numDec == UINT32_MAX ? 4 : numDec;
179 
180  switch (mType) {
181  case DataType::S16:
182  case DataType::S32:
183  case DataType::S64:
184  if (numDec > 0 && numDec < 8) {
185  bool wasNegative = false;
186  int64_t val = valueInt();
187  if (val < 0) {
188  val *= -1;
189  wasNegative = true;
190  }
191  string valStr = fmt::format("{}", val);
192 
193  while (valStr.size() < (numDec + 1)) { valStr.insert(0, "0"); }
194  valStr.insert(valStr.size() - numDec, ".");
195 
196  if (wasNegative) { valStr.insert(0, "-"); }
197  return valStr;
198  }
199  return fmt::format("{}", valueInt());
200  case DataType::U16:
201  case DataType::U32:
202  case DataType::U64:
203  if (numDec > 0 && numDec < 8) {
204  string valStr = fmt::format("{}", valueUInt());
205  while (valStr.size() < (numDec + 1)) { valStr.insert(0, "0"); }
206  valStr.insert(valStr.size() - numDec, ".");
207  return valStr;
208  }
209  return fmt::format("{}", valueUInt());
210  default: return "<UNKNOWN TYPE>";
211  }
212 
213  default: break;
214  }
215 
216  // Should never be executed
217  return to_string(valueUInt());
218 }
