36 log::LOGGER log::initialize(std::vector<spdlog::sink_ptr> _sinks) {
37  auto logger = spdlog::get(SMA_MODBUS_SMA_LOGGER_NAME);
38  if (logger) {
39  logger->info("Logger '{}' was already initialized", SMA_MODBUS_SMA_LOGGER_NAME);
40  return logger;
41  }
42 
43  if (!_sinks.empty()) {
44  logger = make_shared<spdlog::logger>(SMA_MODBUS_SMA_LOGGER_NAME, begin(_sinks), end(_sinks));
45  register_logger(logger);
46  } else {
47  logger = spdlog::stdout_color_mt(SMA_MODBUS_SMA_LOGGER_NAME);
48  }
49 
50  return logger;
51 }
