25 inline size_t fnv1aHash(const char *data) {
26  size_t hash = FNV1A_BASE;
27  while (*data != 0) {
28  hash ^= static_cast<size_t>(*(data++));
29  hash *= FNV1A_PRIME;
30  }
31  return hash;
32 }

34 constexpr size_t fnv1aHash(const char *data, size_t n) {
35  size_t hash = FNV1A_BASE;
36  for (size_t i = 0; i < n; ++i) {
37  hash ^= static_cast<size_t>(data[i]);
38  hash *= FNV1A_PRIME;
39  }
40  return hash;
41 }
