53 ErrorCode DataBase::connect() {
54  auto logger = log::get();
55  int errorCode = SQLITE_OK;
56  fs::path filePath(mPath);
57 
58  if (!fs::exists(filePath)) {
59  logger->error("DataBase::connect() [{}]: DB does not exist", mPath);
60  return ErrorCode::FILE_NOT_FOUND;
61  }
62 
63  if (!fs::is_regular_file(filePath)) {
64  logger->error("DataBase::connect() [{}]: DB is not a regular file", mPath);
65  return ErrorCode::FILE_NOT_FOUND;
66  }
67 
68  logger->debug("DataBase::connect() [{}]: Loading register DB", mPath);
69  errorCode = sqlite3_open(mPath.c_str(), &mDB);
70  if (errorCode != SQLITE_OK) {
71  logger->error("DataBase::connect() [{}]: unable to open the database", mPath);
72  logger->error(" -- Error: '{}'", sqlite3_errstr(errorCode));
73  return ErrorCode::DATA_BASE_ERROR;
74  }
75 
76  if (!validate()) {
77  logger->error("DataBase::connect() [{}]: validation failed", mPath);
78  disconnect();
79  return ErrorCode::DATA_BASE_ERROR;
80  }
81 
82  return ErrorCode::OK;
83 }

95 bool DataBase::validate() {
96  if (!isConnected()) { return false; }
97 
98  auto logger = log::get();
99  logger->debug("Validating '{}':", mPath);
100 
101  vector<string> tables = getTableList();
102  vector<string> missingTables = {};
103  set<string> requiredTables = {"ALL"};
104  auto itTableALL = find(begin(tables), end(tables), "ALL");
105  auto itTableDevEnum = find(begin(tables), end(tables), "DeviceENUM");
106 
107  logger->debug(" -- Total number of tables: {}", tables.size());
108  logger->debug(" - Has table 'ALL': {}", itTableALL != end(tables));
109  logger->debug(" - Has table 'DeviceENUM': {}", itTableDevEnum != end(tables));
110 
111 
112  if (tables.empty() || itTableALL == end(tables) || itTableDevEnum == end(tables)) {
113  logger->error("Important tables missing");
114  return false;
115  }
116 
117  vector<DevEnum> enums = getDeviceEnums();
118  logger->debug(" -- Number of supported devices: {}", enums.size());
119 
120  if (enums.empty()) {
121  logger->error("No entries in table 'DeviceENUM'");
122  return false;
123  }
124 
125  for (auto &i : enums) {
126  if (find(begin(tables), end(tables), i.table) == end(tables)) { missingTables.push_back(i.table); }
127  requiredTables.insert(i.table);
128  }
129 
130  if (!missingTables.empty()) {
131  logger->error("Some tables are missing:");
132  for (auto i : missingTables) { logger->error(" -- table {}", i); }
133  return false;
134  }
135 
136  missingTables = {};
137  uint32_t count = 0;
138  for (auto i : requiredTables) {
139  vector<Register> registers = getRegisters(i);
140 
141  if (registers.empty()) {
142  logger->error("No registers defined in table '{}'", i);
143  missingTables.push_back(i);
144  continue;
145  }
146 
147  count += registers.size();
148  }
149 
150  if (!missingTables.empty()) {
151  logger->error("No registers defined in the following tables:");
152  for (auto i : missingTables) { logger->error(" -- {}", i); }
153  return false;
154  }
155 
156  logger->debug(" -- Found {} register descriptions in {} tables", count, requiredTables.size());
157  return true;
158 }

166 vector<string> DataBase::getTableList() {
167  if (!isConnected()) { return {}; }
168  vector<string> tableList;
169  int errorCode;
170 
171  SQL_Query qTables(mDB, "SELECT `name` FROM `sqlite_master` WHERE `type`='table';");
172  if (!qTables.isValid()) { return {}; }
173 
174  while (true) {
175  errorCode = sqlite3_step(qTables());
176 
177  if (errorCode == SQLITE_DONE) { break; }
178  if (errorCode != SQLITE_ROW) {
179  auto logger = log::get();
180  logger->error("DataBase: getTableList(): error in sqlite3_step:");
181  logger->error(" -- Path: '{}'", mPath);
182  logger->error(" -- Error: '{}'", sqlite3_errstr(errorCode));
183  logger->error(" -- Message: '{}'", sqlite3_errmsg(mDB));
184  return {};
185  }
186 
187  auto *tableName = sqlite3_column_text(qTables(), 0);
188  if (!tableName) {
189  log::get()->error("DataBase: getTableList(): sqlite3_column_text returned NULL (DB: {})", mPath);
190  return {};
191  }
192 
193  tableList.emplace_back((const char *)tableName);
194  }
195 
196  return tableList;
197 }

205 vector<DataBase::DevEnum> DataBase::getDeviceEnums() {
206  if (!isConnected()) { return {}; }
207  vector<DevEnum> enumList;
208  int errorCode;
209 
210  SQL_Query query(mDB, "SELECT `id`, `table`, `name` FROM `DeviceENUM`;");
211  if (!query.isValid()) { return {}; }
212 
213  while (true) {
214  errorCode = sqlite3_step(query());
215 
216  if (errorCode == SQLITE_DONE) { break; }
217  if (errorCode != SQLITE_ROW) {
218  auto logger = log::get();
219  logger->error("DataBase: getDeviceEnums(): error in sqlite3_step:");
220  logger->error(" -- Path: '{}'", mPath);
221  logger->error(" -- Error: '{}'", sqlite3_errstr(errorCode));
222  logger->error(" -- Message: '{}'", sqlite3_errmsg(mDB));
223  return {};
224  }
225 
226  int id = sqlite3_column_int(query(), 0);
227  auto *table = sqlite3_column_text(query(), 1);
228  auto *name = sqlite3_column_text(query(), 2);

230  log::get()->error("DataBase: getDeviceEnums(): sqlite3_column_text returned NULL (DB: {})", mPath);

233 
234  enumList.push_back({(uint32_t)id, (const char *)table, (const char *)name});
235  }

247 vector<Register> DataBase::getRegisters(std::string _table) {
248  if (!isConnected()) { return {}; }
249  vector<Register> regList;
250  int errorCode;
251  auto logger = log::get();
252 
253  SQL_Query query(
254  mDB,
255  fmt::format("SELECT `register`, `desc`, `unit`, `type`, `format`, `access` FROM `{}` ORDER BY `register` ASC;",
256  _table));
257  if (!query.isValid()) { return {}; }
258 
259  while (true) {
260  errorCode = sqlite3_step(query());
261 
262  if (errorCode == SQLITE_DONE) { break; }
263  if (errorCode != SQLITE_ROW) {
264  logger->error("DataBase: getRegisters(_table = '{}'): error in sqlite3_step:", _table);
265  logger->error(" -- Path: '{}'", mPath);
266  logger->error(" -- Error: '{}'", sqlite3_errstr(errorCode));
267  logger->error(" -- Message: '{}'", sqlite3_errmsg(mDB));
268  return {};
269  }
270 
271  int rawID = sqlite3_column_int(query(), 0);
272  auto *rawDesc = sqlite3_column_text(query(), 1);
273  auto *rawUnit = sqlite3_column_text(query(), 2);
274  auto *rawType = sqlite3_column_text(query(), 3);
275  auto *rawFormat = sqlite3_column_text(query(), 4);
276  auto *rawAccess = sqlite3_column_text(query(), 5);
277  if (!rawDesc || !rawUnit || !rawType || !rawFormat || !rawAccess) {
278  logger->error("DataBase: getRegisters(_table = '{}'): sqlite3_column_text returned NULL", _table);
279  return {};
280  }
281 
282  uint16_t id = (uint16_t)rawID;
283  string desc = (const char *)rawDesc;
284  string unit = (const char *)rawUnit;
285  string strType = (const char *)rawType;
286  string strFormat = (const char *)rawFormat;
287  string strAccess = (const char *)rawAccess;
288  DataType type = enum2Str::typeFromStr(strType);
289  DataFormat format = enum2Str::formatFromStr(strFormat);
290  DataAccess access = enum2Str::accessFromStr(strAccess);
291 
292  if (type == DataType::__UNKNOWN__) {
293  logger->error("DataBase: getRegisters(_table = '{}'): Unkown data type '{}'", _table, strType);
294  logger->error(" -- ID: {}", id);
295  logger->error(" -- Description: {}", desc);
296  return {};
297  }
298 
299  if (format == DataFormat::__UNKNOWN__) {
300  logger->error("DataBase: getRegisters(_table = '{}'): Unkown data format '{}'", _table, strFormat);
301  logger->error(" -- ID: {}", id);
302  logger->error(" -- Description: {}", desc);
303  return {};
304  }
305 
306  if (access == DataAccess::__UNKNOWN__) {
307  logger->error("DataBase: getRegisters(_table = '{}'): Unkown access type '{}'", _table, strAccess);
308  logger->error(" -- ID: {}", id);
309  logger->error(" -- Description: {}", desc);
310  return {};
311  }
312 
313  regList.emplace_back(id, desc, unit, type, format, access);
314  }
315 
316  return regList;
317 }
