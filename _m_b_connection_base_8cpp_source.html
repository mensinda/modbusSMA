33 ErrorCode MBConnectionBase::connect() {
34  auto logger = log::get();
35  if (mConnection) { disconnect(); }
36 
37  mConnection = createModbusContext();
38  if (!mConnection) { return ErrorCode::INVALID_MODBUS_CONTEXT; }
39 
40  logger->debug("MBConnectionBase: Establishing the modbus connection");
41  if (modbus_connect(mConnection) == -1) {
42  logger->error("MBConnectionBase: Failed to establish the modbus connection: '{}'", modbus_strerror(errno));
43  modbus_close(mConnection);
44  modbus_free(mConnection);
45  mConnection = nullptr;
46  return ErrorCode::MODBUS_CONNECTION_FAILED;
47  }
48 
49  return ErrorCode::OK;
50 }

72 vector<uint16_t> MBConnectionBase::readRegisters(uint32_t _reg, uint32_t _num) {
73  if (_num > SMA_MODBUS_MAX_REGISTER_COUNT) {
74  auto logger = log::get();
75  logger->error("MBConnectionBase: readRegisters(_reg = {}, _num = {}): ", _reg, _num);
76  logger->error(" -- Can not request {} registers. Max register count is {}", _num, SMA_MODBUS_MAX_REGISTER_COUNT);
77  return {};
78  }
79 
80  vector<uint16_t> vecOut;
81  vecOut.resize(_num);
82 
83  if (modbus_read_registers(mConnection, _reg, _num, vecOut.data()) < 0) {
84  auto logger = log::get();
85  logger->error("MBConnectionBase: readRegisters(_reg = {}, _num = {}): ", _reg, _num);
86  logger->error(" -- Request failed with '{}'", modbus_strerror(errno));
87  return {};
88  }
89 
90  return vecOut;
91 }
