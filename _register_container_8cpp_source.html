28 vector<Register> RegisterContainer::getRegisters(vector<uint16_t> _regList) {
29  vector<Register> outRegList = {};
30  outRegList.reserve(_regList.size()); // Best case: all registers are found
31 
32  for (uint16_t i : _regList) {
33  auto pos = lower_bound(begin(mRegisters), end(mRegisters), i); // the register vector is sorted
34  if (*pos == i) { outRegList.push_back(*pos); }
35  }
36 
37  return outRegList;
38 }

41 void RegisterContainer::addRegisters(vector<Register> _registers) {
42  mRegisters.insert(end(mRegisters), begin(_registers), end(_registers));
43  stable_sort(begin(mRegisters), end(mRegisters));
44  mRegisters.erase(unique(mRegisters.begin(), mRegisters.end()), mRegisters.end());
45 }

48 bool RegisterContainer::updateRegister(uint16_t _address, vector<uint16_t> _data) {
49  auto pos = lower_bound(begin(mRegisters), end(mRegisters), _address); // the register vector is always sorted
50  if (not(*pos == _address)) { return false; }
51 
52  return pos->setRaw(_data);
